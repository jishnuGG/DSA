public class Solution {
    public ListNode detectCycle(ListNode head) {
        ListNode slow=head;
        ListNode fast=head;
         if(head==null || head.next==null)
       {
        return null;
       }
        while(fast!=null && fast.next!=null)
        {
           slow=slow.next;
           fast=fast.next.next;
           if(fast==slow)
           {
            ListNode p1=head;
            ListNode p2=slow;
             while(p1!=p2)
           {
            p1=p1.next;
            p2=p2.next;
            }
            return p1;
           }
        }
        return null;
    }
}

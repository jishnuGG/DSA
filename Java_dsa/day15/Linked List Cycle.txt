import java.util.*;
public class Solution {
    public boolean hasCycle(ListNode head) {
         HashSet<ListNode> map=new HashSet<>();
        ListNode temp=head;
        int flag=0;
        while(temp!=null)
        {
            if(map.contains(temp))
            {
               flag=1;
               break;
            }
            else{
                map.add(temp);
               temp=temp.next;
            }
        }
        if(flag==1)
        {
            return true;
        }
        else{
            return false;
        }
    }
}
